function exampleLet() {
  let name = "Tumiso";
  let age = 25;
  document.getElementById("outputLet").innerText = `Name: ${name}, Age: ${age}`;
}

function exampleConst() {
  const PI = 3.14159;
  document.getElementById("outputConst").innerText = `Value of PI: ${PI}`;
}

function exampleReassign() {
  let counter = 0;
  counter = counter + 1;
  document.getElementById("outputReassign").innerText = `Counter after update: ${counter}`;
}

function exampleMix() {
  let firstName = "Tumiso";
  let score = 95;
  let message = firstName + " scored " + score + "% in JavaScript!";
  document.getElementById("outputMix").innerText = message;
}

function runUserCode() {
  const code = document.getElementById("codeEditor").value;
  try {
    const result = eval(code);
    document.getElementById("outputUserCode").innerText = result !== undefined ? result : "Code ran successfully!";
  } catch (err) {
    document.getElementById("outputUserCode").innerText = "Error: " + err.message;
  }
}
